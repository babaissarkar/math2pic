% Implementing the idea of TeX stackexchange question 34054

% A template inside which the input tex file with
% the equation to be converted will be inserted
% (via the \input{} macro),
% and then this file will be compiled by latex.

% Needs to be compiled with the -shell-escape option to pdflatex/xelatex

\documentclass[convert={density=150, outext=.jpg}]{standalone}
\usepackage{amsmath, amssymb}
\usepackage{varwidth}

\begin{document}
\begin{varwidth}{\linewidth}
	\input{input.tex}	
\end{varwidth}
\end{document}
